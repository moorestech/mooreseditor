# Plugin Contract Consolidation Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Eliminate plugin contract drift by moving shared dependency, manifest/build artifact, persistence, and plugin path validation contracts into host/SDK-owned code with tests.

**Architecture:** The plugin SDK becomes the single source of truth for build-time plugin contracts: shared dependencies, Vite externals, generated `plugin.json`, and small ergonomics helpers. The host owns all runtime validation and persistence semantics: import maps are generated, plugin file paths are boundary-checked, manifest metadata is verified, and `saveProject` either stops claiming atomicity or is implemented through one host-owned persistence module with clearly documented transaction semantics.

**Tech Stack:** TypeScript, React 19, Vite 6, Vitest, Tauri FS plugin, workspace package exports.

```ts
import type { Plugin, UserConfig } from "vite";

import { sharedDependencySpecs } from "./sharedDeps";

export interface MooresPluginMetadata {
  id: string;
  name: string;
  version: string;
}

export interface PluginJson extends MooresPluginMetadata {
  entry: "dist/index.js";
  styles: ["dist/index.css"];
}

export function createPluginJson(metadata: MooresPluginMetadata): PluginJson {
  return {
    ...metadata,
    entry: "dist/index.js",
    styles: ["dist/index.css"],
  };
}

export function pluginManifestPlugin(metadata: MooresPluginMetadata): Plugin {
  return {
    name: "mooreseditor-plugin-manifest",
    generateBundle() {
      this.emitFile({
        type: "asset",
        fileName: "../plugin.json",
        source: `${JSON.stringify(createPluginJson(metadata), null, 2)}\n`,
      });
    },
  };
}

export function mooresPlugin(metadata: MooresPluginMetadata): UserConfig {
  return {
    plugins: [pluginManifestPlugin(metadata)],
    build: {
      lib: {
        entry: "src/plugin-entry.tsx",
        formats: ["es"],
        fileName: () => "index.js",
      },
      rollupOptions: {
        external: sharedDependencySpecs(),
        output: {
          inlineDynamicImports: true,
          assetFileNames: "index[extname]",
        },
      },
      outDir: "dist",
      emptyOutDir: true,
    },
  };
}

export { sharedDependencySpecs };
```

Create `apps/mooreseditor/src/pluginHost/pluginPaths.ts`:

```ts
function normalizeSlashes(value: string): string {
  return value.replace(/\\/g, "/");
}

export function resolvePluginPackagePath(
  pluginDir: string,
  packageRelativePath: string,
): string {
  const normalizedRelative = normalizeSlashes(packageRelativePath);
  if (
    normalizedRelative.trim() === "" ||
    normalizedRelative.startsWith("/") ||
    /^[a-zA-Z]:\//.test(normalizedRelative)
  ) {
    throw new Error(`Invalid plugin file path: ${packageRelativePath}`);
  }

  const segments = normalizedRelative.split("/").filter(Boolean);
  if (segments.includes("..")) {
    throw new Error(`Invalid plugin file path: ${packageRelativePath}`);
  }

  const normalizedDir = normalizeSlashes(pluginDir).replace(/\/+$/, "");
  return `${normalizedDir}/${segments.join("/")}`;
}
```

```ts
export function injectPluginStyleLink(pluginId: string, href: string): void {
  const selector = `link[data-plugin-id="${CSS.escape(pluginId)}"][href="${CSS.escape(href)}"]`;
  if (document.querySelector(selector)) {
    return;
  }

  const link = document.createElement("link");
  link.rel = "stylesheet";
  link.href = href;
  link.dataset.pluginId = pluginId;
  link.dataset.pluginStyle = "true";
  document.head.appendChild(link);
}

export function removePluginStyleLinks(pluginId: string): void {
  document
    .querySelectorAll(`link[data-plugin-id="${CSS.escape(pluginId)}"][data-plugin-style="true"]`)
    .forEach((node) => node.remove());
}
```

- [ ] **Step 1: Decide transaction contract before editing**

Use this exact contract unless Tauri/Rust transaction support is added in the same task:

```ts
/**
 * Saves master columns and optional plugin files as one host-owned save request.
 *
 * The host validates every target before writing and rejects the whole request before
 * the first write if any target path is invalid. Files are then written sequentially.
 * This is not a filesystem-atomic transaction across multiple files.
 */
```

Reason: JavaScript-side Tauri FS has `writeTextFile`, but this codebase does not currently expose a Rust command for temp-write plus atomic rename/replace. Removing the false atomic promise is the minimal contract fix. A later Rust command can upgrade the implementation to true atomic directory-level commit.

```ts
function validateRelativePath(relativePath: string): void {
  if (relativePath.startsWith("/")) {
    throw new Error(`Invalid relativePath (must not be absolute): ${relativePath}`);
  }
  const segments = relativePath.split("/").filter((s) => s.length > 0);
  if (segments.length === 0) {
    throw new Error(`Invalid relativePath (empty): ${relativePath}`);
  }
  if (segments.includes("..")) {
    throw new Error(`Invalid relativePath (path traversal '..' not allowed): ${relativePath}`);
  }
}

async function resolveAndPrepareDir(
  projectDir: string,
  relativePath: string,
): Promise<string> {
  const segments = relativePath.split("/").filter((s) => s.length > 0);
  const dirSegments = segments.slice(0, -1);
  let currentDir = projectDir;

  for (const segment of dirSegments) {
    currentDir = await path.join(currentDir, segment);
    if (segment.startsWith(".")) {
      try {
        await invoke("add_project_to_scope", { projectPath: currentDir });
      } catch {
        // dev/browser fallback path
      }
    }
    if (!(await exists(currentDir))) {
      await mkdir(currentDir, { recursive: true });
    }
  }

  return path.join(currentDir, ...segments.slice(-1));
}

export async function saveProjectFiles({
  projectDir,
  masterDir,
  columns,
  extraFiles = [],
}: SaveProjectFilesParams): Promise<void> {
  if (!projectDir) {
    throw new Error("saveProject: projectDir is not set");
  }
  if (!masterDir) {
    throw new Error("saveProject: masterDir is not set");
  }

  for (const file of extraFiles) {
    validateRelativePath(file.path);
  }

  for (const column of columns) {
    const jsonFilePath = await path.join(masterDir, `${column.title}.json`);
    await writeTextFile(jsonFilePath, JSON.stringify(column.data, null, 2));
  }

  for (const file of extraFiles) {
    const targetPath = await resolveAndPrepareDir(projectDir, file.path);
    await writeTextFile(targetPath, file.content);
  }
}
```

```ts
import { readFileSync } from "node:fs";

import {
  SHARED_DEPENDENCIES,
  sharedDependencySpecs,
} from "@mooreseditor/plugin-sdk/vite";

function readJson(path: string): Record<string, unknown> {
  return JSON.parse(readFileSync(path, "utf8")) as Record<string, unknown>;
}

const hostPackage = readJson("apps/mooreseditor/package.json");
const sdkPackage = readJson("packages/plugin-sdk/package.json");
const nodeGraphPackage = readJson("plugins/node-graph/package.json");

const sharedSpecs = new Set(sharedDependencySpecs());

for (const dep of SHARED_DEPENDENCIES) {
  if (dep.spec === "react/jsx-runtime") {
    continue;
  }
  const hostVersion = (hostPackage.dependencies as Record<string, string>)[dep.spec];
  if (hostVersion !== dep.versionRange) {
    throw new Error(`Host dependency ${dep.spec} is ${hostVersion}, expected ${dep.versionRange}`);
  }
}

for (const pkg of [sdkPackage, nodeGraphPackage]) {
  const peerDependencies = (pkg.peerDependencies ?? {}) as Record<string, string>;
  for (const spec of sharedSpecs) {
    if (spec === "react/jsx-runtime" || spec === "@xyflow/react") {
      continue;
    }
    const expected = SHARED_DEPENDENCIES.find((dep) => dep.spec === spec)?.versionRange;
    if (peerDependencies[spec] !== expected) {
      throw new Error(`${pkg.name} peer ${spec} is ${peerDependencies[spec]}, expected ${expected}`);
    }
  }
}
```
